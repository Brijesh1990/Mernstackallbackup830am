sql join ... 

             a) inner join 
                examples : 
                   select flipkart.flip_teacher.*,collegename,studentname from flipkart.flip_teacher inner join flipkart.flip_college on flipkart.flip_teacher.collegid=flipkart.flip_college.collegid inner join  flipkart.flip_student on flipkart.flip_teacher.studentid=flipkart.flip_student.studentid
 
             b) join 
               examples :
select flipkart.flip_teacher.*,collegename,studentname from flipkart.flip_teacher  join flipkart.flip_college on flipkart.flip_teacher.collegid=flipkart.flip_college.collegid  join  flipkart.flip_student on flipkart.flip_teacher.studentid=flipkart.flip_student.studentid
 
                   or 

select teacherid,name,phone,address,department,collegename,studentname from flipkart.flip_teacher  join flipkart.flip_college on flipkart.flip_teacher.collegid=flipkart.flip_college.collegid  join  flipkart.flip_student on flipkart.flip_teacher.studentid=flipkart.flip_student.studentid
 
             c) outer join 
                   1) left join 
                         data matched from first table to second table with its left rows if matched                            join otherwise return null values.

                       examples : 
select teacherid,name,phone,address,department,collegename,studentname from flipkart.flip_teacher left join flipkart.flip_college on flipkart.flip_teacher.collegid=flipkart.flip_college.collegid  left join  flipkart.flip_student on flipkart.flip_teacher.studentid=flipkart.flip_student.studentid

                   2) right join 
                          data matched from second table to first table with its right rows if matched                            join otherwise return null values.
                           examples: 
select teacherid,name,phone,address,department,collegename,studentname from flipkart.flip_teacher right join flipkart.flip_college on flipkart.flip_teacher.collegid=flipkart.flip_college.collegid  right join  flipkart.flip_student on flipkart.flip_teacher.studentid=flipkart.flip_student.studentid
 

                   3) full join 
                           full join = left join + right join 
                           note: not supported in mysql

              d) cross join 

                        crosss join join tables with multiply by data;
                        examples : select * from flipkart.flip_teacher cross join flipkart.flip_student 
                          

             Note : join is used to join more than one tables with its common field if data matched                     join if datanot matched return null values.
